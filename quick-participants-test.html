const resultsDiv = document.getElementById('results') const testBtn = document.getElementById('testBtn') testBtn.disabled = true resultsDiv.innerHTML = '
⏳ 正在测试...
' let html = '' try { // 测试1: 检查用户信息 const userInfo = JSON.parse(localStorage.getItem('userInfo') || '{}') if (!userInfo.userId) { html += `
❌ 测试失败：未找到用户信息
请先登录系统
` resultsDiv.innerHTML = html testBtn.disabled = false return } html += `
✅ 用户信息正常
用户ID: ${userInfo.userId}
昵称: ${userInfo.nickName}
` // 测试2: 检查会议信息 const meetingResponse = await fetch(`${API_BASE}/meetingInfo/getCurrentMeeting`, { headers: { 'token': localStorage.getItem('token') } }) const meetingResult = await meetingResponse.json() if (!meetingResult.data || !meetingResult.data.meetingId) { html += `
❌ 测试失败：未找到当前会议
请先创建或加入一个会议
` resultsDiv.innerHTML = html testBtn.disabled = false return } const meetingId = meetingResult.data.meetingId const meetingNo = meetingResult.data.meetingNo html += `
✅ 会议信息正常
会议号: ${meetingNo}
会议ID: ${meetingId}
` // 测试3: 加载成员列表 const membersResponse = await fetch(`${API_BASE}/meetingInfo/loadMeetingMembers?meetingId=${meetingId}`, { headers: { 'token': localStorage.getItem('token') } }) const membersResult = await membersResponse.json() if (membersResult.code !== 200 || !membersResult.data) { html += `
❌ 测试失败：无法加载成员列表
错误: ${membersResult.msg || '未知错误'}
` resultsDiv.innerHTML = html testBtn.disabled = false return } const allMembers = membersResult.data const totalCount = allMembers.length html += `
ℹ️ API返回成员总数: ${totalCount}
` // 测试4: 过滤成员 const currentUserId = userInfo.userId const otherMembers = allMembers.filter(m => m.status === MemberStatus.NORMAL && m.userId !== currentUserId ) const otherCount = otherMembers.length if (otherCount === 0) { html += `
0
⚠️ 会议中没有其他成员
这是正常的！会议中只有你一个人。

如何测试多人显示：
1. 打开另一个浏览器（或隐身窗口）
2. 用另一个账号登录
3. 加入会议号: ${meetingNo}
4. 回到这个页面刷新测试
` } else { html += `
${otherCount}
✅ 成功！会议中有 ${otherCount} 个其他成员
${otherMembers.map(m => ` • ${m.nickName} (ID: ${m.userId})
  头像: ${m.avatar || '使用默认头像'}
  性别: ${m.sex === 1 ? '男' : m.sex === 0 ? '女' : '未设置'} `).join('
')}
` // 检查头像 const membersWithAvatar = otherMembers.filter(m => m.avatar) if (membersWithAvatar.length === 0) { html += `
⚠️ 注意：所有成员都没有设置头像
系统会使用默认头像（根据性别）或生成的头像
` } } // 测试5: 检查状态异常的成员 const abnormalMembers = allMembers.filter(m => m.status !== MemberStatus.NORMAL) if (abnormalMembers.length > 0) { html += `
ℹ️ 发现 ${abnormalMembers.length} 个状态异常的成员（已退出/被踢出）
这些成员不会显示在会议中：
${abnormalMembers.map(m => ` • ${m.nickName} - 状态: ${m.status} `).join('
')}
` } } catch (error) { html += `
❌ 测试出错
${error.message}
` } resultsDiv.innerHTML = html testBtn.disabled = false }
